### Expected Implementation Pattern

Based on the existing DashboardEmitter pattern (verify before implementing):

```javascript
// In DashboardEmitter class:

emitBobPhaseChange(phase, data = {}) {
  this.emit('bob:phase:change', { phase, ...data, timestamp: Date.now() });
}

emitBobAgentSpawned(agentId, data = {}) {
  this.emit('bob:agent:spawned', { agentId, ...data, timestamp: Date.now() });
}

emitBobAgentCompleted(agentId, result = {}) {
  this.emit('bob:agent:completed', { agentId, result, timestamp: Date.now() });
}

emitBobSurfaceDecision(surface, decision) {
  this.emit('bob:surface:decision', { surface, decision, timestamp: Date.now() });
}

emitBobError(error, context = {}) {
  this.emit('bob:error', {
    message: error instanceof Error ? error.message : String(error),
    stack: error instanceof Error ? error.stack : undefined,
    context,
    timestamp: Date.now()
  });
}
```
